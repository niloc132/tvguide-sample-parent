35  /**
36   * @author colin
37   */
38  public class TvGuideServiceModule extends AbstractModule {
39  
40  	@Override
41  	protected void configure() {
42  		// TvViewers will be handed out by this, depending on how auth is scoped
43  		bind(TvViewer.class).toProvider(AuthenticatedViewerProvider.class);
44  
45  		// Make the CallFactories available for injection
46  		// Auth calls
47  		install(new FactoryModuleBuilder().implement(AuthenticationCall.class,
48  				AuthCallable.class).build(AuthenticationCallFactory.class));
49  		// TvGuide calls
50  		install(new FactoryModuleBuilder().implement(EpisodesDateCall.class,
51  				EpisodeRangeCallable.class).implement(UserFavoritesCall.class,
52  				FavoritesChannelCallable.class).implement(
53  				ChannelListCall.class, ChannelListCallable.class).build(
54  				TvGuideCallFactory.class));
55  
56  		//bind(EpisodesDateCall.class).to(EpisodeRangeCallable.class);
57  		// Ensure that something has session stuff ready
58  		requireBinding(AuthenticatedViewerProvider.class);
59  
60  		// Ensure that something is bringing a EntityManager to the party
61  		requireBinding(EntityManager.class);
62  
63  		// TvViewer must be statically injected, otherwise it has no access to the great 
64  		// Guicy world of data
65  		requestStaticInjection(TvViewer.class);
66  	}
67  }
